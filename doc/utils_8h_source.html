235  template<typename T>
236  CORE_API std::vector<T> split(const T& s, const char delim, bool obey_brackets = false)
237  {
238  std::vector<T> result;
239  T item = "";
240 
241  if (obey_brackets)
242  {
243  int bracket_level = 0;
244 
245  for (size_t i = 0; i < s.length(); ++i)
246  {
247  char c = s.at(i);
248  switch (c)
249  {
250  case '(':
251  case '{':
252  case '[':
253  ++bracket_level;
254  break;
255  case ')':
256  case '}':
257  case ']':
258  --bracket_level;
259  break;
260  default:
261  break;
262  }
263  if (bracket_level < 0)
264  {
265  bracket_level = 0;
266  }
267  if (c == delim)
268  {
269  // No constant expression, therefore not usable in switch case
270  if (bracket_level == 0)
271  {
272  result.push_back(item);
273  item = "";
274  }
275  else
276  {
277  item.push_back(c);
278  }
279  }
280  else
281  {
282  item.push_back(c);
283  }
284  }
285  if (!item.empty())
286  {
287  result.push_back(item);
288  }
289  }
290  else
291  {
292  std::stringstream ss(s);
293  while (std::getline(ss, item, delim))
294  {
295  result.push_back(item);
296  }
297  }
298  if (s.back() == delim)
299  {
300  result.push_back("");
301  }
302  return result;
303  }

380  template<typename T>
381  CORE_API T replace(const T& str, const T& search, const T& replace)
382  {
383  auto s = str;
384  size_t pos = 0;
385 
386  if (search.empty())
387  {
388  return str;
389  }
390 
391  while ((pos = s.find(search, pos)) != T::npos)
392  {
393  s.replace(pos, search.length(), replace);
394  pos += replace.length();
395  }
396 
397  return s;
