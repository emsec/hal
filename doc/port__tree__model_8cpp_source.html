338  bool ModulePinsTreeModel::canDropMimeData(const QMimeData *data, Qt::DropAction action, int row, int column, const QModelIndex &parent) const
339  {
340  Q_UNUSED(action);
341  Q_UNUSED(column);
342  if(!data->formats().contains("pintreemodel/item")) return false;
343 
344  QString type; int id;
345  auto encItem = data->data("pintreemodel/item");
346  QDataStream dataStream(&encItem, QIODevice::ReadOnly);
347  dataStream >> type >> id;
348  auto parentItem = static_cast<PortTreeItem*>(getItemFromIndex(parent));
349  qDebug() << "type: " << type << ", id" << id << ", row: " << row;
350 
351  // construct a "drop-matrix" here, but only 4(5) things are NOT allowed (so check for these):
352  // 1: drop a pin on its OWN parent
353  // 3: drop a pingroup on/between pins
354  // 4: drop an item on itself
355  // 5: drop adjecent index to itself, must be at least 1 item apart
356  if(type == "group")
357  {
358  auto item = mIdToGroupItem[id];
359  if(parentItem)
360  {
361  auto type = getTypeOfItem(parentItem);
362  if(type == itemType::pin || (type == itemType::group && row != -1) || item == parentItem)
363  return false;
364  }
365  else // here, only check for adjacent rows
366  {
367  auto itRow = item->getOwnRow();
368  if(itRow == row || ((itRow+1) == row))
369  return false;
370  }
371  }
372  if(type == "pin")
373  {
374  // perhaps check here that a pin can only be dropped between groups if its own group has size > 1?
375  // otherwise it does not make much sense...perhaps change this check
376  auto item = mIdToPinItem[id];
377  if((!parentItem && item->getParent()->getChildCount() == 1) || (item->getParent() == parentItem && row == -1) || item == parentItem
378  || (parentItem && (getTypeOfItem(parentItem) == itemType::pin)))
379  return false;
380  // case if one wants to drop between pins in same group, check if its not adjacent row (other cases are handled on case above
381  if(item->getParent() == parentItem)
382  {
383  auto itRow = item->getOwnRow();
384  if(itRow == row || ((itRow+1) == row))
385  return false;
386  }
387  }
388  return true;
389  }
