256  // takes a map of unconditional transitions and reconstructs the conditions under which each condition is taken
257  Result<std::map<u64, std::map<u64, BooleanFunction>>> generate_conditional_transitions(const std::vector<std::pair<Net*, BooleanFunction>>& state_bfs,
258  const std::map<u64, std::set<u64>>& transitions)
259  {
260  // generate all transitions that are reachable from the inital state.
261  std::map<u64, std::map<u64, BooleanFunction>> conditional_transitions;
262 
263  // for all possible and previously found successor states we build the condition to reach them
264  for (const auto& [prev, successors] : transitions)
265  {
266  // this builds a mapping for all the output net variables of the state vector to the current starting state
267  std::map<std::string, BooleanFunction> prev_mapping;
268  for (u32 i = 0; i < state_bfs.size(); i++)
269  {
270  prev_mapping.insert(
271  {BooleanFunctionNetDecorator(*(state_bfs.at(i).first)).get_boolean_variable_name(), ((prev >> i) & 1) ? BooleanFunction::Const(1, 1) : BooleanFunction::Const(0, 1)});
272  }
273 
274  for (const auto& suc : successors)
275  {
276  // this all the boolean functions of incoming data nets to the state vector either vanilla incase the corresponding state bit is 1 in the successor state or negated incase the state bit is 0 in the successor state.
277  BooleanFunction condition;
278 
279  for (u32 i = 0; i < state_bfs.size(); i++)
280  {
281  auto next_state_bit_bf = ((suc >> i) & 1) ? state_bfs.at(i).second : BooleanFunction::Not(state_bfs.at(i).second.clone(), 1).get();
282 
283  if (condition.is_empty())
284  {
285  condition = next_state_bit_bf;
286  }
287  else
288  {
289  condition = BooleanFunction::And(std::move(condition), std::move(next_state_bit_bf), 1).get();
290  }
291  }
292 
293  // replace all the variables of the previous state with their real values for our current state n and simplify.
294  condition = condition.substitute(prev_mapping).get();
295 
296  // we are left with a condition that only includes the inputs to the fsm that needs to be fullfilled to reach the successor state from state n.
297  condition = condition.simplify();
298 
299  conditional_transitions[prev].insert({suc, condition});
300  }
301  }
302 
303  return OK(conditional_transitions);
304  }

326  BooleanFunction next_state_vec = state_bfs.front().second;
327  for (u32 i = 1; i < state_reg.size(); i++)

333 
334  for (u64 state = 0; state < (u64(1) << state_size); state++)
335  {

337  std::map<std::string, BooleanFunction> var_to_val;

339  {
340  std::string var = BooleanFunctionNetDecorator(*(state_bfs.at(state_index).first)).get_boolean_variable_name();
341  BooleanFunction val = ((state >> state_index) & 0x1) ? BooleanFunction::Const(1, 1) : BooleanFunction::Const(0, 1);
342  var_to_val.insert({var, val});
343  }
344

353

360  {
361  std::string input_var = inputs.at(input_index);

363  input_mapping.insert({input_var, {val}});
364  }

373

377  }

381  }

393  }

395  /* END DEBUG PRINTING */

400  }

405  Result<std::map<u64, std::map<u64, BooleanFunction>>> SolveFsmPlugin::solve_fsm(Netlist* nl,

413  if (state_size > 64)
414  {
415  return ERR("failed to solve fsm: Currently only supports fsm with up to 64 state flip-flops but got " + std::to_string(state_size) + ".");
416  }

419  const auto state_bfs_res = generate_state_bfs(nl, state_reg, transition_logic, true);

425 
426  BooleanFunction prev_state_vec = BooleanFunctionNetDecorator(*(state_bfs.front().first)).get_boolean_variable();
427  BooleanFunction next_state_vec = state_bfs.front().second;
428  for (u32 i = 1; i < state_reg.size(); i++)
429  {
430  // bitvector representing the previous state
431  prev_state_vec = BooleanFunction::Concat(BooleanFunctionNetDecorator(*(state_bfs.at(i).first)).get_boolean_variable(), std::move(prev_state_vec), i + 1).get();
432 
433  // bitvector including all the functions to calculate the next state
434  next_state_vec = BooleanFunction::Concat(state_bfs.at(i).second.clone(), std::move(next_state_vec), i + 1).get();
435  }
436 
437  // generate initial state
438  u64 initial_state_num = 0;
439  if (!initial_state.empty())
440  {
441  for (const auto& gate : state_reg)
442  {
443  if (initial_state.find(gate) == initial_state.end())
444  {
445  return ERR("failed to solve fsm: Unable to find intial value for gate " + std::to_string(gate->get_id()) + " in the provided initial state map.");
446  }
447

449  initial_state_num += initial_state.at(gate);
450  }
451  }
452 
453  // generate all transitions that are reachable from the inital state.
454  std::map<u64, std::set<u64>> all_transitions;
455 
456  std::deque<u64> q;
457  std::unordered_set<u64> visited;
458 
459  q.push_back(initial_state_num);
460 
461  while (!q.empty())
462  {
463  std::vector<u64> successor_states;
464 
465  u64 n = q.front();
466  q.pop_front();
467 
468  if (visited.find(n) != visited.end())
469  {

471  }

475  SMT::Solver s;
476 
477  // set prev_state_vec to starting state
478  s = s.with_constraint(SMT::Constraint{prev_state_vec.clone(), BooleanFunction::Const(n, state_size)});
479 
480  while (true)
481  {
482  if (auto res = s.query(SMT::QueryConfig().with_model_generation().with_timeout(timeout)); res.is_error())
483  {
484  return ERR_APPEND(res.get_error(), "failed to solve fsm: failed to querry SMT solver for state " + std::to_string(n) + ".");
485  }
486  else
487  {
488  auto s_res = res.get();
489 
490  if (s_res.is_unsat())
491  {
492  break;
493  }
494 
495  if (s_res.is_unknown())
496  {
497  return ERR("failed to solve fsm: received an unknown solver result for state " + std::to_string(n) + ".");
498  }
499 
500  auto m = s_res.model.value();
501  auto suc = m.evaluate(next_state_vec).get();
502  auto suc_num = 0;
503 
504  // a constant (numeral) successor state
505  if (suc.is_constant())
506  {
507  suc_num = suc.get_constant_value_u64().get();
508  }
509  // a successor state that includes boolean functions (for example in form of input variables)
510  else
511  {
512  // to resolve such a successor state, we simpply set all variables left in the state to zero (which is one possible solution) and continue to search for more valid solutions
513  std::unordered_map<std::string, std::vector<BooleanFunction::Value>> zero_mapping;
514  for (const auto& var : suc.get_variable_names())
515  {
516  zero_mapping.insert({var, {BooleanFunction::Value::ZERO}});
517  }
518 
519  if (auto eval_res = suc.evaluate(zero_mapping); eval_res.is_error())
520  {
521  return ERR_APPEND(eval_res.get_error(), "failed to solve fsm: could not evaluate successor state to constant.");
522  }
523  else
524  {
525  suc_num = BooleanFunction::to_u64(eval_res.get()).get();
526  }
527  }
528 
529  q.push_back(suc_num);
530  all_transitions[n].insert(suc_num);
531  s = s.with_constraint(SMT::Constraint(BooleanFunction::Not(BooleanFunction::Eq(next_state_vec.clone(), BooleanFunction::Const(suc_num, suc.size()), 1).get(), 1).get()));
532  }
533  }
534  }
535 
536  const auto conditional_transitions = generate_conditional_transitions(state_bfs, all_transitions).get();
537 
538  /* DEBUG PRINTING */
539  for (const auto& [org, successors] : conditional_transitions)
540  {
541  std::cout << org << ": " << std::endl;
542  for (const auto& [suc, condition] : successors)
543  {
544  std::cout << "\t" << suc << ": " << condition.to_string() << std::endl;
545  }
546  }
547  /* END DEBUG PRINTING */
548 
549  if (auto graph_res = generate_dot_graph(state_reg, conditional_transitions, graph_path); graph_res.is_error())
550  {
551  return ERR_APPEND(graph_res.get_error(), "failed to solve fsm: unable to generate dot graph.");
552  }
553 
554  return OK(conditional_transitions);
