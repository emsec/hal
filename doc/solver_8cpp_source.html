202  {
203  auto translate_declarations = [] (const std::vector<Constraint>& _constraints) -> std::string {
204  std::set<std::tuple<std::string, u16>> inputs;
205  for (const auto& constraint : _constraints) {
206  for (const auto& node : constraint.lhs.get_reverse_polish_notation()) {
207  if (auto variable = node->get_as<BooleanFunction::OperandNode>(); node->is_variable()) {
208  inputs.insert(std::make_tuple(variable->variable, node->size));
209  }
210  }
211  }
212 
213  return std::accumulate(inputs.begin(), inputs.end(), std::string(),
214  [] (auto accumulator, auto entry) -> std::string {
215  return accumulator + "(declare-fun " + std::get<0>(entry) + " () (_ BitVec " + std::to_string(std::get<1>(entry)) + "))\n";
216  }
217  );
218  };

221  return std::accumulate(_constraints.begin(), _constraints.end(), std::make_tuple(true, std::string()),
222  [] (auto state , auto constraint) -> std::tuple<bool, std::string> {
