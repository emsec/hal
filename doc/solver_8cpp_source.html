453  Result<std::string> Solver::translate_to_smt2(const std::vector<Constraint>& constraints, const QueryConfig& config)
454  {
459  auto translate_declarations = [](const std::vector<Constraint>& _constraints) -> std::string {
460  std::set<std::tuple<std::string, u16>> inputs;
461  for (const auto& constraint : _constraints)
462  {
463  if (constraint.is_assignment())
464  {
465  for (const auto& node : constraint.get_assignment().get()->first.get_nodes())
466  {
467  if (node.is_variable())
468  {
469  inputs.insert(std::make_tuple(node.variable, node.size));
470  }
471  }
472  }
473  else
474  {
475  for (const auto& node : constraint.get_function().get()->get_nodes())
476  {
477  if (node.is_variable())
478  {
479  inputs.insert(std::make_tuple(node.variable, node.size));
480  }
481  }
482  }
483  }
484 
485  return std::accumulate(inputs.begin(), inputs.end(), std::string(), [](auto accumulator, auto entry) -> std::string {
486  return accumulator + "(declare-fun " + std::get<0>(entry) + " () (_ BitVec " + std::to_string(std::get<1>(entry)) + "))\n";
487  });
488  };
489 
494  auto translate_constraints = [](const std::vector<Constraint>& _constraints) -> Result<std::string> {
495  return std::accumulate(_constraints.cbegin(), _constraints.cend(), Result<std::string>::Ok({}), [](auto accumulator, const auto& constraint) -> Result<std::string> {
496  // (1) short-hand termination in case accumulator is an error
497  if (accumulator.is_error())
498  {
499  return ERR(accumulator.get_error());
500  }
501 
502  if (constraint.is_assignment())
503  {
504  const auto assignment = constraint.get_assignment().get();
505  auto lhs = Translator::translate_to_smt2(assignment->first);
506  auto rhs = Translator::translate_to_smt2(assignment->second);
507  if (lhs.is_error())
508  {
509  return ERR_APPEND(lhs.get_error(), "could not translate constraint to SMT-LIB v2: '" + constraint.to_string() + "'");
510  }
511  else if (rhs.is_error())
512  {
513  return ERR_APPEND(rhs.get_error(), "could not translate constraint to SMT-LIB v2: '" + constraint.to_string() + "'");
514  }
515  else
516  {
517  return OK(accumulator.get() + "(assert (= " + lhs.get() + " " + rhs.get() + "))\n");
518  }
519  }
520  else
521  {
522  auto lhs = Translator::translate_to_smt2(*constraint.get_function().get());
523  if (lhs.is_ok())
524  {
525  return OK(accumulator.get() + "(assert (= #b1 " + lhs.get() + "))\n");
526  }
527  return ERR_APPEND(lhs.get_error(), "could not translate constraint to SMT-LIB v2: '" + constraint.to_string() + "'");
528  }
529  });
530  };
531 
532  auto theory = std::string("(set-logic QF_ABV)");
533  auto declarations = translate_declarations(constraints);
534  auto constraints_str = translate_constraints(constraints);
535  auto epilogue = std::string("(check-sat)") + ((config.generate_model) ? "\n(get-model)" : "");
536 
537  if (constraints_str.is_ok())
538  {
539  return OK(theory + "\n" + declarations + "\n" + constraints_str.get() + "\n" + epilogue);
540  }
541  return ERR_APPEND(constraints_str.get_error(), "could not translate constraint to SMT-LIB v2: unable to generate translation constraints");
542  }
