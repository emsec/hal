713  {
714  return OK(p[0]);
715  }
716 
717  // X | 1 => 1
718  if (p[1] == One(node.size))
719  {
720  return OK(p[1]);
721  }
722 
723  // X | X => X
724  if (p[0] == p[1])
725  {
726  return OK(p[0]);
727  }
728 
729  // X | ~X => 111...1
730  if ((~p[0] == p[1]) || (p[0] == ~p[1]))
731  {
732  return OK(One(node.size));
733  }
734 
735  if (p[0].is(BooleanFunction::NodeType::And) && p[1].is(BooleanFunction::NodeType::And))
736  {
737  auto p0_parameter = p[0].get_parameters();
738  auto p1_parameter = p[1].get_parameters();
739 
740  // (X & Y) | (X & Z) => X & (Y | Z)
741  if (p0_parameter[0] == p1_parameter[0])
742  {
743  return OK(p0_parameter[0] & (p0_parameter[1] | p1_parameter[1]));
744  }
745  // (X & Y) | (Z & X) => X & (Y | Z)
746  if (p0_parameter[0] == p1_parameter[1])
747  {
748  return OK(p0_parameter[0] & (p0_parameter[1] | p1_parameter[0]));
749  }
750  // (X & Y) | (Y & Z) => Y & (Y | Z)
751  if (p0_parameter[1] == p1_parameter[0])
752  {
753  return OK(p0_parameter[1] & (p0_parameter[0] | p1_parameter[1]));
754  }
755  // (X & Y) | (Z & Y) => Y & (X | Z)
756  if (p0_parameter[1] == p1_parameter[1])
757  {
758  return OK(p0_parameter[1] & (p0_parameter[0] | p1_parameter[0]));
759  }
760  }
761 
762  if (p[1].is(BooleanFunction::NodeType::And))
763  {
764  auto p1_parameter = p[1].get_parameters();
765  // X | (Y & !X) => X | Y
766  if ((~p1_parameter[1]) == p[0])
767  {
768  return BooleanFunction::Or(p[0].clone(), p1_parameter[0].clone(), node.size);
769  }
770 
771  // X | (X & Y) => X
772  if ((p1_parameter[0] == p[0]) || (p1_parameter[1] == p[0]))
773  {
774  return OK(p[0]);
775  }
776

778  if ((~p1_parameter[0] == p[0]) || (p1_parameter[0] == ~p[0]))
779  {
780  return BooleanFunction::Or(p[0].clone(), p1_parameter[1].clone(), node.size);
781  }
782  // X | (Y & ~X) => X | Y
783  if ((~p1_parameter[1] == p[0]) || (p1_parameter[1] == ~p[0]))
784  {
785  return BooleanFunction::Or(p[0].clone(), p1_parameter[0].clone(), node.size);
786  }
787  }
788 
789  if (p[1].is(BooleanFunction::NodeType::Or))
790  {
791  auto p1_parameter = p[1].get_parameters();
792

799  if (p1_parameter[1] == p[0])

808  }

812  {
813  return OK(One(node.size));
814  }
815  }
816 
817  if (p[0].is(BooleanFunction::NodeType::Or))
818  {

820

1060  }

1083  }
1084  }

1189  case BooleanFunction::NodeType::Ite: {

1191  if (p[0].has_constant_value(0))

1196  if (p[0].has_constant_value(1))

1198  return OK(p[1]);

1210  }

1213  Result<BooleanFunction> SymbolicExecution::constant_propagation(const BooleanFunction::Node& node, std::vector<BooleanFunction>&& p)
1214  {
1215  if (node.get_arity() != p.size())
1216  {
1217  return ERR("could not propagate constants: arity does not match number of parameters");
1218  }
1219 
1220  std::vector<std::vector<BooleanFunction::Value>> values;
1221  for (const auto& parameter : p)
1222  {
1223  values.emplace_back(parameter.get_top_level_node().constant);
1224  }
1225 
1226  switch (node.type)
1227  {
1228  case BooleanFunction::NodeType::And:
1229  return OK(ConstantPropagation::And(values[0], values[1]));
1230  case BooleanFunction::NodeType::Or:
1231  return OK(ConstantPropagation::Or(values[0], values[1]));
1232  case BooleanFunction::NodeType::Not:
1233  return OK(ConstantPropagation::Not(values[0]));
1234  case BooleanFunction::NodeType::Xor:
1235  return OK(ConstantPropagation::Xor(values[0], values[1]));
1236 
1237  case BooleanFunction::NodeType::Add:
1238  return OK(ConstantPropagation::Add(values[0], values[1]));
1239  case BooleanFunction::NodeType::Sub:
1240  return OK(ConstantPropagation::Sub(values[0], values[1]));
1241  case BooleanFunction::NodeType::Mul:
1242  return OK(ConstantPropagation::Mul(values[0], values[1]));
1243 
1244  case BooleanFunction::NodeType::Slice: {
1245  auto start = p[1].get_index_value().get();
1246  auto end = p[2].get_index_value().get();
1247  return OK(BooleanFunction::Const(std::vector<BooleanFunction::Value>(values[0].begin() + start, values[0].begin() + end)));
1248  }
1249  case BooleanFunction::NodeType::Concat: {
1250  values[1].insert(values[1].end(), values[0].begin(), values[0].end());
1251  return OK(BooleanFunction::Const(values[1]));
1252  }
1253  case BooleanFunction::NodeType::Zext: {
1254  values[0].resize(node.size, BooleanFunction::Value::ZERO);
1255  return OK(BooleanFunction::Const(values[0]));
1256  }
1257  case BooleanFunction::NodeType::Sext: {
1258  values[0].resize(node.size, static_cast<BooleanFunction::Value>(values[0].back()));
1259  return OK(BooleanFunction::Const(values[0]));
1260  }
1261 
1262  case BooleanFunction::NodeType::Eq:
1263  return OK((values[0] == values[1]) ? BooleanFunction::Const(1, 1) : BooleanFunction::Const(0, 1));
1264  case BooleanFunction::NodeType::Sle:
1265  return OK(ConstantPropagation::Sle(values[0], values[1]));
1266  case BooleanFunction::NodeType::Slt:
1267  return OK(ConstantPropagation::Slt(values[0], values[1]));
1268  case BooleanFunction::NodeType::Ule:
1269  return OK(ConstantPropagation::Ule(values[0], values[1]));
1270  case BooleanFunction::NodeType::Ult:
1271  return OK(ConstantPropagation::Ult(values[0], values[1]));
1272  case BooleanFunction::NodeType::Ite:
1273  return OK(ConstantPropagation::Ite(values[0], values[1], values[2]));
1274
