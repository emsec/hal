524  GraphContext* GraphContextManager::restoreFromFile(const QString& filename)
525  {
526  QFile jsFile(filename);
527  if (!jsFile.open(QIODevice::ReadOnly))
528  return nullptr;
529  QJsonDocument jsonDoc = QJsonDocument::fromJson(jsFile.readAll());
530  const QJsonObject& json = jsonDoc.object();
531  GraphContext* firstContext = nullptr;
532  GraphContext* selectedContext = nullptr;
533 
534  if (json.contains("views") && json["views"].isArray())
535  {
536  QJsonArray jsonViews = json["views"].toArray();
537  int nviews = jsonViews.size();
538  for (int iview = 0; iview < nviews; iview++)
539  {
540  QJsonObject jsonView = jsonViews.at(iview).toObject();
541  if (!jsonView.contains("id") || !jsonView["id"].isDouble())
542  continue;
543  u32 viewId = jsonView["id"].toInt();
544  if (!jsonView.contains("name") || !jsonView["name"].isString())
545  continue;
546  QString viewName = jsonView["name"].toString();
547  if (viewId > mMaxContextId)
548  mMaxContextId = viewId;
549  GraphContext* context = gGraphContextManager->getContextById(viewId);
550  if (context)
551  {
552  // load view in existing context
553  context->clear();
554  if (viewName != context->mName)
555  renameGraphContextAction(context,viewName);
556  if (!context->readFromFile(jsonView))
557  {
558  log_warning("gui", "failed to set up existing context ID={} from view file {}.", context->id(), filename.toStdString());
559  continue;
560  }
561  }
562  else
563  {
564  // create new context
565  context = new GraphContext(viewId, viewName);
566  context->setLayouter(getDefaultLayouter(context));
567  context->setShader(getDefaultShader(context));
568  // context->setExclusiveModuleId(exclusiveModuleId); TODO exclusive module ID
569  context->scene()->setDebugGridEnabled(mSettingDebugGrid->value().toBool());
570  connect(mSettingDebugGrid, &SettingsItemCheckbox::boolChanged, context->scene(), &GraphicsScene::setDebugGridEnabled);
571 
572  if (!context->readFromFile(jsonView))
573  {
574  log_warning("gui", "failed to create context context ID={} view file {}.", context->id(), filename.toStdString());
575  deleteGraphContext(context);
576  continue;
577  }
578 
579  mContextTableModel->beginInsertContext(context);
580  mContextTableModel->addContext(context);
581  mContextTableModel->endInsertContext();
582  Q_EMIT contextCreated(context);
583  }
584 
585  if (jsonView.contains("exclusiveModuleId"))
586  context->setExclusiveModuleId(jsonView["exclusiveModuleId"].toInt(),false);
587  if (jsonView.contains("selected"))
588  selectedContext = context;
589  if (!firstContext)
590  firstContext = context;
591  context->setDirty(false);
592  }
593  }
594  if (selectedContext) return selectedContext;
595  return firstContext;
596  }
