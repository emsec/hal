632  GraphContext* GraphContextManager::restoreFromFile(const QString& filename)
633  {
634  QFile jsFile(filename);
635  if (!jsFile.open(QIODevice::ReadOnly))
636  return nullptr;
637  QJsonDocument jsonDoc = QJsonDocument::fromJson(jsFile.readAll());
638  const QJsonObject& json = jsonDoc.object();
639  GraphContext* firstContext = nullptr;
640  GraphContext* selectedContext = nullptr;
641 
642  if (json.contains("directories") && json["directories"].isArray())
643  {
644  QJsonArray jsonDirectories = json["directories"].toArray();
645  int ndir = jsonDirectories.size();
646  for (int idir = 0; idir < ndir; idir++)
647  {
648  QJsonObject jsondir = jsonDirectories.at(idir).toObject();
649  if (!jsondir.contains("id") || !jsondir["id"].isDouble())
650  continue;
651  u32 dirId = jsondir["id"].toInt();
652  if (!jsondir.contains("name") || !jsondir["name"].isString())
653  continue;
654  QString dirName = jsondir["name"].toString();
655  if (!jsondir.contains("parentId"))
656  continue;
657  u32 dirParentId = jsondir["parentId"].toInt();
658 
659  BaseTreeItem* dirParent = mContextTreeModel->getRootItem();
660 
661  if (dirParentId != 0)
662  dirParent = mContextTreeModel->getDirectory(dirParentId);
663 
664  if (dirId < 0 || dirId > 0x7FFFFFFF)
665  dirId = ++mMaxContextId;
666 
667  mContextTreeModel->addDirectory(dirName, dirParent, dirId);
668  }
669  }
670  if (json.contains("views") && json["views"].isArray())
671  {
672  QJsonArray jsonViews = json["views"].toArray();
673  int nviews = jsonViews.size();
674  for (int iview = 0; iview < nviews; iview++)
675  {
676  QJsonObject jsonView = jsonViews.at(iview).toObject();
677  if (!jsonView.contains("id") || !jsonView["id"].isDouble())
678  continue;
679  u32 viewId = jsonView["id"].toInt();
680  if (!jsonView.contains("name") || !jsonView["name"].isString())
681  continue;
682  QString viewName = jsonView["name"].toString();
683  if (viewId > mMaxContextId)
684  mMaxContextId = viewId;
685  int visibleFlag = 1; // default to visible before flag was invented
686  if (jsonView.contains("visible"))
687  visibleFlag = jsonView["visible"].toInt();
688 
689  u32 viewParentId = 0;
690  if (!jsonView.contains("parentId"))
691  viewParentId = jsonView["parentId"].toInt();
692 
693  BaseTreeItem* viewParent = mContextTreeModel->getRootItem();
