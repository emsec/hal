170  std::tuple<bool, Model> Model::parse(std::string s, const SolverType& type)
171  {
172  // TODO:
173  // check how to attach a local parser context variable to the x3 parser
174  // in order to prevent the necessity of a mutex and thus speed-up
175  // parallel processing.
176 
177  // to transform the current parser implementation into a thread-safe
178  // implementation, we use a static mutex variable that is locked-upon
179  // entry in order to prevent any overwrite of the parser_context data
180  static std::mutex mutex;
181 
182  // (1) reset current context in a thread-safe way
183  std::lock_guard<std::mutex> lock(mutex);
184  ModelParser::parser_context = ModelParser::ParserContext();
185 
186  // (2) parse model using the SMT-LIB grammars for the different solver
187  auto iter = s.begin();
188  auto ok = [&]() -> bool {
189  switch (type)
190  {
191  case SolverType::Z3:
192  return boost::spirit::x3::phrase_parse(iter, s.end(), ModelParser::Z3_MODEL_GRAMMAR, boost::spirit::x3::space);
193  case SolverType::Boolector:
194  return boost::spirit::x3::phrase_parse(iter, s.end(), ModelParser::BOOLECTOR_MODEL_GRAMMAR, boost::spirit::x3::space);
195 
196  default: return false;
197  }
198  }();
199 
200  switch (ok && (iter == s.end()))
201  {
202  case true: return {true, Model(ModelParser::parser_context.model)};
203  default: return {false, Model()};
