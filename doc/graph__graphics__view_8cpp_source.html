1195  void GraphGraphicsView::addSuccessorToView(int maxLevel, bool succ)
1196  {
1197 
1198  QSet<u32> gatsNew;
1199  QSet<const Gate*> gatsHandled;
1200 
1201  QList<const Gate*> startList;
1202  Node startNode;
1203 
1204  switch (mItem->itemType())
1205  {
1206  case ItemType::Gate:
1207  startNode = Node(mItem->id(),Node::Gate);
1208  startList.append(gNetlist->get_gate_by_id(mItem->id()));
1209  break;
1210  case ItemType::Module:
1211  startNode = Node(mItem->id(),Node::Module);
1212  for (const Gate* g : gNetlist->get_module_by_id(mItem->id())->get_gates(nullptr,true))
1213  startList.append(g);
1214  break;
1215  default:
1216  return;
1217  }
1218 
1219  Q_ASSERT(startList.size());
1220 
1221  for (const Gate* g : startList)
1222  gatsHandled.insert(g);
1223 
1224  const NodeBoxes& boxes = mGraphWidget->getContext()->getLayouter()->boxes();
1225  const NodeBox* box = boxes.boxForNode(startNode);
1226  Q_ASSERT(box);
1227  int xOrigin = box->x();
1228  int yOrigin = box->y();
1229  int xDir = succ ? 1 : -1;
1230 
1231  PlacementHint plc(PlacementHint::Standard);
1232 
1233  for (int loop = 0; !maxLevel || loop<maxLevel; loop++)
1234  {
1235  int y = 0;
1236  QList<const Gate*> foundList;
1237 
1238  for (const Gate* gOrigin : startList)
1239  {
1240  for (const Gate* g : netlist_utils::get_next_gates(gOrigin, succ, 1))
1241  {
1242  if (gatsHandled.contains(g)) continue;
1243  gatsHandled.insert(g);
1244  if (boxes.boxForGate(g)) continue; // by in view
1245  foundList.append(g);
1246  }
1247  }
1248 
1249  if (foundList.isEmpty()) break;
1250  for (const Gate* g: foundList)
1251  {
1252  gatsNew.insert(g->get_id());
1253  QPoint point(xOrigin + (loop+1) * xDir, yOrigin + y);
1254  y = y > 0 ? -y : -y+1;
1255  Node nd(g->get_id(),Node::Gate);
1256  if (!boxes.boxForPoint(point)) // not occupied yet
1257  {
1258  if (plc.mode() == PlacementHint::Standard)
1259  plc = PlacementHint(PlacementHint::GridPosition);
1260  plc.addGridPosition(nd,point);
1261  }
1262  }
1263  startList = foundList;

1843  GraphGraphicsView::LayouterPoint GraphGraphicsView::closestLayouterPoint(qreal scene_pos, int default_spacing, int min_index, QVector<qreal> sections) const
1844  {

1847  if (sections.first() > scene_pos)
1848  {
1849  int distance = sections.first() - scene_pos;
1850  int nSections = distance / default_spacing; // this rounds down
1851  index -= (nSections + 1);
1852  pos = sections.first() + index * default_spacing;
1853  }
1854  else if (sections.last() <= scene_pos)
1855  {
1856  int distance = scene_pos - sections.last();
1857  int nSections = distance / default_spacing; // this rounds down
1858  index += (sections.size() + nSections -1);
1859  pos = sections.last() + nSections * default_spacing;
1860  }
1861  else
1862  {

1864  auto it = sections.constBegin();
1865  auto jt = it+1;
1866  while (jt != sections.constEnd())
1867  {
1868  if (scene_pos < *jt) break; // found value in inteval [it..[jt
1869  ++it;

1871  ++index;
1872  }
1873  pos = *it;
1874  }
