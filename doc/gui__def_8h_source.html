145  QPoint pos = QPoint(gatePosition(gateId)->first, gatePosition(gateId)->second); //position of current gate to move
146  hal::Node nd = key(QPoint(p.first, p.second)); //find the node in the destination
147 
148  if(!nd.isNull() && !swap) //if the destination placement is not available
149  log_warning("gui", "Target position is already occupied");
150  else if (!nd.isNull() && swap)
151  {
152  operator[](hal::Node(gateId,hal::Node::Gate)) = QPoint(p.first,p.second);//set the position of the first node to the destination
153  if(nd.isGate())
154  operator[](hal::Node(nd.id(), hal::Node::Gate)) = pos;//set the position of the destination node to the position of the first node
155  else operator[](hal::Node(nd.id(), hal::Node::Module)) = pos;
156  }
157  else
158  operator[](hal::Node(gateId,hal::Node::Gate)) = QPoint(p.first,p.second);
159  }

162  hal::Node nd = key(QPoint(p.first, p.second));
163 
164  if(!nd.isNull() && !swap)
165  log_warning("gui", "Target position is already occupied");
166  else if (!nd.isNull() && swap)

169  if(nd.isGate())
