69 package body _sha_256_pkg is
70 - function ROTR (a : std_logic_vector(WORD_SIZE1 downto 0); n : natural)
71  return std_logic_vector is
72  --result : std_logic_vector(WORD_SIZE-1 downto 0);
73  begin
74  --signal result : std_logic_vector(WORD_SIZE-1 downto 0);
75  return (std_logic_vector(shift_right(unsigned(a), n))) or std_logic_vector((shift_left(unsigned(a), (WORD_SIZE-n))));
76  end function;
77 
78 - function ROTL (a : std_logic_vector(WORD_SIZE1 downto 0); n : natural)
79  return std_logic_vector is
80  --result : std_logic_vector(WORD_SIZE-1 downto 0);
81  begin
82  --signal result : std_logic_vector(WORD_SIZE-1 downto 0);
83  return (std_logic_vector(shift_left(unsigned(a), n))) or std_logic_vector((shift_right(unsigned(a), (WORD_SIZE-n))));
84  end function;
85 
86 - function SHR (a : std_logic_vector(WORD_SIZE1 downto 0); n : natural)
87  return std_logic_vector is
88  begin
89  return std_logic_vector(shift_right(unsigned(a), n));
90  end function;

92 --- function CH (x : std_logic_vector(WORD_SIZE1 downto 0);
93  y : std_logic_vector(WORD_SIZE1 downto 0);
94  z : std_logic_vector(WORD_SIZE1 downto 0))
95  return std_logic_vector is
96  begin
97  return (x and y) xor (not(x) and z);
