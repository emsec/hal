223  // be no way back

563  void GraphContext::applyChanges()
564  {
565  // since changes are only applied once in a while added module might not exist any more
566  auto it = mAddedModules.begin();
567  while (it != mAddedModules.end())
568  {
569  if (gNetlist->get_module_by_id(*it))
570  ++it;
571  else
572  it = mAddedModules.erase(it);
573  }
574 
575  mModules -= mRemovedModules;
576  mGates -= mRemovedGates;
577 
578  mModules += mAddedModules;
579  mGates += mAddedGates;
580 
581  mLayouter->remove(mRemovedModules, mRemovedGates, mNets);
582  mShader->remove(mRemovedModules, mRemovedGates, mNets);
583 
584  updateNets();
585 
586  int placementOrder[4] = { PlacementHint::GridPosition,
587  PlacementHint::PreferLeft,
588  PlacementHint::PreferRight,
589  PlacementHint::Standard};
590 
591  for (int iplc = 0; iplc<4; iplc++)
592  {
593  for (const PlacementEntry& plcEntry : mPlacementList[placementOrder[iplc]])
594  {
595  QSet<u32> modsForHint = plcEntry.mModules;
596  modsForHint &= mAddedModules; // may contain obsolete entries that we must filter out
597  QSet<u32> gatsForHint = plcEntry.mGates;
598  gatsForHint &= mAddedGates;
599  mLayouter->add(modsForHint, gatsForHint, mNets, plcEntry.mPlacementHint);
600  }
601  }
602 
603  mShader->add(mAddedModules, mAddedGates, mNets);
604 
605  mAddedModules.clear();
606  mAddedGates.clear();
607 
608  mRemovedModules.clear();
609  mRemovedGates.clear();
610 
611  for (int i=0; i<4; i++)
612  mPlacementList[i].clear();
613 
614  mUnappliedChanges = false;
615  mSceneUpdateRequired = true;
616  }
