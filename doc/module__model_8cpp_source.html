422  void ModuleModel::findNetParentRecursion(BaseTreeItem* parent, QHash<const Net *, ModuleItem *> &parentAssignment, std::unordered_set<Net*>& assignedNets) const
423  {

449  Module* ModuleModel::findNetParent(const Net *net) const
450  {
451  QHash<Module*,int> modHash;
452  if (net->is_global_input_net() || net->is_global_output_net()) return nullptr;
453  int maxDepth = 0;
454 
455  for (const Endpoint* ep : net->get_sources())
456  {
457  Module* m = ep->get_gate()->get_module();
458  Q_ASSERT(m);
459  int depth = m->get_submodule_depth();
460  if (depth > maxDepth) maxDepth = depth;
461  modHash.insert(m,depth);
462  }
463 
464  for (const Endpoint* ep : net->get_destinations())
465  {
466  Module* m = ep->get_gate()->get_module();
467  Q_ASSERT(m);
468  int depth = m->get_submodule_depth();
469  if (depth > maxDepth) maxDepth = depth;
470  modHash.insert(m,depth);
471  }
472 
473  while (modHash.size() > 1 && maxDepth > 0)
474  {
475  auto it = modHash.begin();
476  while (it != modHash.end())
477  {
478  if (it.value() == maxDepth)
479  {
480  Module* parentMod = it.key()->get_parent_module();
481  modHash.erase(it);
482  if (parentMod) modHash.insert(parentMod,maxDepth-1);
483  break;
484  }
485  ++it;
486  }
487  if (it == modHash.end())
488  --maxDepth;
489  }
490  if (modHash.empty()) return nullptr;
491  return modHash.begin().key();
492  }
