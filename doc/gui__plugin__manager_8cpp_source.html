279  void GuiPluginTable::populateTable(bool refresh)
280  {
281  QDir userConfigDir(QString::fromStdString(utils::get_user_config_directory()));
282  QMap<QString,GuiPluginEntry*> pluginEntries;
283  mSettings = new QSettings(userConfigDir.absoluteFilePath("plugins.ini"),QSettings::IniFormat);
284 
285  std::set<std::string> loadedPlugins = plugin_manager::get_plugin_names();
286 
287  int nentry = mSettings->beginReadArray("plugins");
288  for (int i=0; i<nentry; i++)
289  {
290  mSettings->setArrayIndex(i);
291  GuiPluginEntry* gpe = new GuiPluginEntry(mSettings);
292  pluginEntries.insert(gpe->mName,gpe);
293  }
294  mSettings->endArray();
295 
296  for (std::filesystem::path plugDir : utils::get_plugin_directories())
297  {
298  QDir dir(QString::fromStdString(plugDir.string()));
299  if (!dir.exists()) continue;
300  for (QFileInfo info : dir.entryInfoList())
301  {
302  if (!info.isFile()) continue;
303  if (!plugin_manager::has_valid_file_extension(info.absoluteFilePath().toStdString())) continue;
304 
305  QString pluginName = info.baseName();
306  GuiPluginEntry* gpe = pluginEntries.value(pluginName);
307  bool needUpdate = true;
308  if (gpe)
309  {
310  needUpdate = gpe->mFileModified != info.lastModified();
311  gpe->setFileFound(true);
312  }
313  else
314  {
315  gpe = new GuiPluginEntry(info);
316  pluginEntries.insert(pluginName,gpe);
317  }
318 
319  BasePluginInterface* bpif = plugin_manager::get_plugin_instance(pluginName.toStdString(), true, true);
320  if (bpif && !refresh)
321  {
322  // plugin already loaded ?
323  bool alreadyLoaded = loadedPlugins.find(pluginName.toStdString())!=loadedPlugins.end();
324  gpe->updateFromLoaded(bpif, alreadyLoaded);
325  log_debug("gui", "GuiPluginManager: '{}' already loaded.", pluginName.toStdString());
326  continue;
327  }
328  else if (gpe->requestLoad())
329  {
330  // user requested this plugin explicitly in previous session
331  bpif = load(pluginName, info.absoluteFilePath());
332  if (!bpif)
333  {
334  gpe->mState = GuiPluginEntry::NotAPlugin; // however, load failed
335  log_warning("gui", "GuiPluginManager: loading of requested plugin '{}' failed.", pluginName.toStdString());
336  }
337  else
338  {
339  gpe->updateFromLoaded(bpif, true, info.lastModified());
340  log_info("gui", "GuiPluginManager: '{}' loaded upon user request.", pluginName.toStdString());
341  }
342  continue;
343  }
344 
345  if (!needUpdate)
346  {
347  log_debug("gui", "GuiPluginManager: '{}' not modified, no update required.", pluginName.toStdString());
348  // known plugin, load only when needed
349  if (gpe->isPlugin())
350  gpe->mState = GuiPluginEntry::NotLoaded;
351  continue;
352  }
353 
354  // plugin updated or not known yet, try dummy load to retrieve information
355  bpif = load(pluginName, info.absoluteFilePath());
356  if (!bpif)
357  {
358  log_warning("gui", "GuiPluginManager: loading of '{}' failed, is it really a HAL plugin?", pluginName.toStdString());
359  gpe->mState = GuiPluginEntry::NotAPlugin;
360  continue;
361  }
362 
363  if (bpif->get_name().empty() || bpif->get_version().empty())
364  {
365  log_warning("gui", "GuiPluginManager: '{}' has empty name or version, plugin ignored.", pluginName.toStdString());
366  gpe->mState = GuiPluginEntry::NotAPlugin;
367  }
368  else
369  {
370  log_info("gui", "GuiPluginManager: '{}' loaded to retrieve information about plugin features.", pluginName.toStdString());
371  gpe->updateFromLoaded(bpif, false, info.lastModified()); // load success
372  }
373  }
374  }
375 
376  // update entry if plugin loaded in meantime as dependency
377  for (std::string loadedPluginName: plugin_manager::get_plugin_names())
378  {
379  GuiPluginEntry* gpe = pluginEntries.value(QString::fromStdString(loadedPluginName));
380  if (!gpe || gpe->isLoaded()) continue;
381  BasePluginInterface* bpif = plugin_manager::get_plugin_instance(loadedPluginName,false,true);
382  if (!bpif) continue;
383  QFileInfo info(gpe->mFilePath);
384  gpe->updateFromLoaded(bpif,false,info.exists()?info.lastModified():QDateTime());
385  }
386 
387  mEntries.clear();
388  mAvoid.clear();
389  mLookup.clear();
390  auto it = pluginEntries.begin();
391  while (it != pluginEntries.end())
392  {
393  GuiPluginEntry* gpe = it.value();
394  if (gpe->isFileFound())
395  {
396  if (gpe->isPlugin())
397  {
398  mLookup.insert(gpe->mName,mEntries.size());
399  mEntries.append(gpe);
400  }
401  else
402  {
403  mAvoid.append(gpe);
404  }
405  ++it;
406  }
407  else
408  it = pluginEntries.erase(it);
409  }
410  persist();
411  }
