224  void NetlistElementsTreeModel::moduleSubmoduleRemoved(Module *m, int removed_module)
225  {
226  //1. go through the actual TreeItems through a BFS and remove them from the maps
227  //2. delete the associated module tree items (beginResetModel)
228  Module* removedMod = gNetlist->get_module_by_id(removed_module);
229  QList<TreeItem*> tmpSubmodItems; // they already get removed in the BFS
230  for(TreeItem* removedSubmodItem : mModuleToTreeitems.values(removedMod))
231  {
232  tmpSubmodItems.append(removedSubmodItem);
233  QQueue<TreeItem*> treeItemsQueue;
234  treeItemsQueue.enqueue(removedSubmodItem);
235  while(!treeItemsQueue.isEmpty())
236  {
237  TreeItem* currentItem = treeItemsQueue.dequeue();
238  int id = currentItem->getAdditionalData(keyRepresentedID).toUInt();
239 
240  switch (getTypeOfItem(currentItem))
241  {
242  case itemType::module: mModuleToTreeitems.remove(gNetlist->get_module_by_id(id),currentItem); break;
243  case itemType::gate: mGateToTreeitems.remove(gNetlist->get_gate_by_id(id), currentItem); break;
244  case itemType::net: mNetToTreeitems.remove(gNetlist->get_net_by_id(id), currentItem); break;
245  }
246 
247  for(TreeItem* child : currentItem->getChildren())
248  treeItemsQueue.enqueue(child);
249  }
250  }
251  //after clearing the maps, delete the corresponding module items (propagates through all children)
252  beginResetModel();
253  for(TreeItem* removedSubItem : tmpSubmodItems)
254  {
255  removedSubItem->getParent()->removeChild(removedSubItem);
256  if(mNetsDisplayed)
257  updateInternalNetsOfModule(removedSubItem->getParent());
258  delete removedSubItem;
259  }
260  endResetModel();
261 
262  if((int)m->get_id() == mModId)
263  Q_EMIT numberOfSubmodulesChanged(m->get_submodules().size());
264 
265  if(removed_module == mModId)
266  clear();
267  }
