1708  inline void Child::execute_child() {
1709 #ifndef __USING_WINDOWS__
1710  int sys_ret = -1;
1711  auto& stream = parent_->stream_;
1712 
1713  try {
1714  if (stream.write_to_parent_ == 0)
1715  stream.write_to_parent_ = dup(stream.write_to_parent_);
1716 
1717  if (stream.err_write_ == 0 || stream.err_write_ == 1)
1718  stream.err_write_ = dup(stream.err_write_);
1719 
1720  // Make the child owned descriptors as the
1721  // stdin, stdout and stderr for the child process
1722  auto _dup2_ = [](int fd, int to_fd) {
1723  if (fd == to_fd) {
1724  // dup2 syscall does not reset the
1725  // CLOEXEC flag if the descriptors
1726  // provided to it are same.
1727  // But, we need to reset the CLOEXEC
1728  // flag as the provided descriptors
1729  // are now going to be the standard
1730  // input, output and error
1731  util::set_clo_on_exec(fd, false);
1732  } else if(fd != -1) {
1733  int res = dup2(fd, to_fd);
1734  if (res == -1) throw OSError("dup2 failed", errno);
1735  }
1736  };
1737 
1738  // Create the standard streams
1739  _dup2_(stream.read_from_parent_, 0); // Input stream
1740  _dup2_(stream.write_to_parent_, 1); // Output stream
1741  _dup2_(stream.err_write_, 2); // Error stream
1742 
1743  // Close the duped descriptors
1744  if (stream.read_from_parent_ != -1 && stream.read_from_parent_ > 2)
