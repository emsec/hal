59

63  if (ctx.pass_counter >= num_passes)
64  {
65  ctx.done = true;
66  }
67  }

72 
73  if (auto it = ctx.pass_outcome.find({current_state, current_pass.id}); it != ctx.pass_outcome.end())
74  {
