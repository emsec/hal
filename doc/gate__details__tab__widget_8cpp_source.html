265  void GateDetailsTabWidget::setupBooleanFunctionTables(Gate* gate, GateDetailsTabWidget::GateTypeCategory gateTypeCategory)
266  {
267  if (gate == nullptr)
268  {
269  return;
270  }
271 
272  static QSet<QString> ffBfNames = {
273  "clear",
274  "preset", // Both
275  "clock",
276  "clocked_on",
277  "clocked_on_also",
278  "next_state",
279  "power_down_function" // FF names
280  };
281 
282  static QSet<QString> latchBfNames = {
283  "clear",
284  "preset", // Both
285  "enable",
286  "data_in" // Latch names
287  };
288 
289  static QSet<QString> noSpecialBfNames = {};
290 
291  QSet<QString>* specialBfNames = &noSpecialBfNames;
292  if (gateTypeCategory == GateDetailsTabWidget::GateTypeCategory::ff)
293  {
294  specialBfNames = &ffBfNames;
295  }
296  else if (gateTypeCategory == GateDetailsTabWidget::GateTypeCategory::latch)
297  {
298  specialBfNames = &latchBfNames;
299  }
300 
301  std::unordered_map<std::string, BooleanFunction> allBfs = gate->get_boolean_functions(false);
302 
303  QMap<QString, BooleanFunction> specialFunctions;
304  QMap<QString, BooleanFunction> otherFunctions;
305 
306  for (auto& it : allBfs)
307  {
308  QString bfName = QString::fromStdString(it.first);
309  if (specialBfNames->contains(bfName))
310  {
311  // Function is a LUT/FF function
312  specialFunctions.insert(bfName, it.second);
313  }
314  else
315  {
316  // Function is not a LUT/FF function
317  otherFunctions.insert(bfName, it.second);
318  }
319  }
320 
321  QVector<QSharedPointer<BooleanFunctionTableEntry>> specialFunctionList;
322  QVector<QSharedPointer<BooleanFunctionTableEntry>> otherFunctionList;
323  QVector<QSharedPointer<BooleanFunctionTableEntry>> setPresetBehavior;
324 
325  QMap<QString, BooleanFunction>::iterator i;
326 
327  for (i = specialFunctions.begin(); i != specialFunctions.end(); i++)
328  specialFunctionList.append(QSharedPointer<BooleanFunctionTableEntry>(new BooleanFunctionEntry(gate->get_id(), i.key(), i.value())));
329 
330  for (i = otherFunctions.begin(); i != otherFunctions.end(); i++)
331  otherFunctionList.append(QSharedPointer<BooleanFunctionTableEntry>(new BooleanFunctionEntry(gate->get_id(), i.key(), i.value())));
332 
333  GateType* gt = gate->get_type();
334  if (FFComponent* ff_component = gt->get_component_as<FFComponent>([](const GateTypeComponent* c) { return FFComponent::is_class_of(c); }); ff_component != nullptr)
335  {
336  if (ff_component->get_async_set_reset_behavior().first != AsyncSetResetBehavior::undef)
337  setPresetBehavior.append(QSharedPointer<BooleanFunctionTableEntry>(new FFComponentEntry(gate->get_id(), ff_component->get_async_set_reset_behavior())));
338  if (!ff_component->get_clock_function().is_empty())
339  setPresetBehavior.append(QSharedPointer<BooleanFunctionTableEntry>(new FFComponentEntry(gate->get_id(), FFComponentEntry::FFCompFunc::Clock, ff_component->get_clock_function())));
340  if (!ff_component->get_next_state_function().is_empty())
341  setPresetBehavior.append(
342  QSharedPointer<BooleanFunctionTableEntry>(new FFComponentEntry(gate->get_id(), FFComponentEntry::FFCompFunc::NextState, ff_component->get_next_state_function())));
343  if (!ff_component->get_async_set_function().is_empty())
344  setPresetBehavior.append(
345  QSharedPointer<BooleanFunctionTableEntry>(new FFComponentEntry(gate->get_id(), FFComponentEntry::FFCompFunc::AsyncSet, ff_component->get_async_set_function())));
346  if (!ff_component->get_async_reset_function().is_empty())
347  setPresetBehavior.append(
348  QSharedPointer<BooleanFunctionTableEntry>(new FFComponentEntry(gate->get_id(), FFComponentEntry::FFCompFunc::AsyncReset, ff_component->get_async_reset_function())));
349  }
350  else if (LatchComponent* latch_component = gt->get_component_as<LatchComponent>([](const GateTypeComponent* c) { return LatchComponent::is_class_of(c); }); latch_component != nullptr)
351  {
352  if (latch_component->get_async_set_reset_behavior().first != AsyncSetResetBehavior::undef)
353  setPresetBehavior.append(QSharedPointer<BooleanFunctionTableEntry>(new LatchComponentEntry(gate->get_id(), latch_component->get_async_set_reset_behavior())));
354  if (!latch_component->get_enable_function().is_empty())
355  setPresetBehavior.append(
356  QSharedPointer<BooleanFunctionTableEntry>(new LatchComponentEntry(gate->get_id(), LatchComponentEntry::LatchCompFunc::Enable, latch_component->get_enable_function())));
357  if (!latch_component->get_data_in_function().is_empty())
358  setPresetBehavior.append(
359  QSharedPointer<BooleanFunctionTableEntry>(new LatchComponentEntry(gate->get_id(), LatchComponentEntry::LatchCompFunc::DataInFunc, latch_component->get_data_in_function())));
360  if (!latch_component->get_async_set_function().is_empty())
361  setPresetBehavior.append(
362  QSharedPointer<BooleanFunctionTableEntry>(new LatchComponentEntry(gate->get_id(), LatchComponentEntry::LatchCompFunc::AsyncSet, latch_component->get_async_set_function())));
363  if (!latch_component->get_async_reset_function().is_empty())
364  setPresetBehavior.append(
365  QSharedPointer<BooleanFunctionTableEntry>(new LatchComponentEntry(gate->get_id(), LatchComponentEntry::LatchCompFunc::AsyncReset, latch_component->get_async_reset_function())));
366  }
367 
368  if (StateComponent* state_component = gt->get_component_as<StateComponent>([](const GateTypeComponent* c) { return StateComponent::is_class_of(c); }); state_component != nullptr)
369  {
370  if (state_component->get_state_identifier() != "")
371  setPresetBehavior.append(QSharedPointer<BooleanFunctionTableEntry>(
372  new StateComponentEntry(gate->get_id(), StateComponentEntry::StateCompType::PosState, QString::fromStdString(state_component->get_state_identifier()))));
373  if (state_component->get_neg_state_identifier() != "")
374  setPresetBehavior.append(QSharedPointer<BooleanFunctionTableEntry>(
375  new StateComponentEntry(gate->get_id(), StateComponentEntry::StateCompType::NegState, QString::fromStdString(state_component->get_neg_state_identifier()))));
376  }
377 
378  // Fill the category (LUT/FF/LATCH) widgets
379  switch (gateTypeCategory)
380  {
381  case GateDetailsTabWidget::GateTypeCategory::lut: {
382  const std::vector<std::string> lutPins = gate->get_type()->get_pin_names([](const GatePin* p) { return p->get_type() == PinType::lut; });
383  // LUT Boolean Function Table only shows the LUT function
384  QVector<QSharedPointer<BooleanFunctionTableEntry>> lutEntries;
385  for (auto bfEntry : otherFunctionList)
386  {
387  if (std::find(lutPins.begin(), lutPins.end(), bfEntry->getEntryIdentifier().toStdString()) != lutPins.end())
388  {
389  lutEntries.append(bfEntry);
390  }
391  }
392 
393  mLutFunctionTable->setEntries(lutEntries);
394  mLutFunctionTable->setGateInformation(gate);
395 
396  //Setup lut config (init) string
397  if (InitComponent* init_component = gt->get_component_as<InitComponent>([](const GateTypeComponent* c) { return InitComponent::is_class_of(c); }); init_component != nullptr)
398  {
399  auto typeAndValue = gate->get_data(init_component->get_init_category(), init_component->get_init_identifiers()[0]);
400  mLutConfigLabel->setText(" 0x" + QString::fromStdString(std::get<1>(typeAndValue)));
401  }
402  else
403  {
404  mLutConfigLabel->setText(" Could not load init string.");
405  }
406 
407  // The table is only updated if the gate has a LUT pin
408  if (lutPins.size() > 0)
409  {
410  // All LUT pins have the same boolean function
411  std::basic_string<char> outPin = lutPins.front();
412 
413  // Fill the LUL truth table
414  BooleanFunction lutFunction = gate->get_boolean_function(outPin);
415  mLutTable->setBooleanFunction(lutFunction, QString::fromStdString(outPin));
416  }
417  break;
418  }
419  case GateDetailsTabWidget::GateTypeCategory::ff: {
420  mFfFunctionTable->setEntries(specialFunctionList + setPresetBehavior);
421  break;
422  }
423  case GateDetailsTabWidget::GateTypeCategory::latch: {
424  mLatchFunctionTable->setEntries(specialFunctionList + setPresetBehavior);
425  break;
426  }
427  default:
428  break;
429  }
430  mFullFunctionTable->setEntries(specialFunctionList + otherFunctionList);
431  }
