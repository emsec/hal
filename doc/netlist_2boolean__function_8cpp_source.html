1108  Result<std::vector<std::vector<BooleanFunction::Value>>> BooleanFunction::compute_truth_table(const std::vector<std::string>& ordered_variables, bool remove_unknown_variables) const
1109  {
1110  auto variable_names_in_function = this->get_variable_names();
1111 
1112  // (1) check that each variable is just a single bit, otherwise we do
1113  // not generate a truth-table
1114  for (const auto& node : this->m_nodes)
1115  {
1116  if (node.is_variable() && node.size != 1)
1117  {
1118  return ERR("could not compute truth table for Boolean function '" + this->to_string() + "': unable to generate a truth-table for Boolean function with variables of > 1-bit");
1119  }
1120  }
1121 
1122  // (2) select either parameter or the Boolean function variables
1123  auto variables = ordered_variables;
1124  if (variables.empty())
1125  {
1126  variables = std::vector<std::string>(variable_names_in_function.begin(), variable_names_in_function.end());
1127  }
1128 
1129  // (3) remove any unknown variables from the truth table
1130  if (remove_unknown_variables)
1131  {
1132  variables.erase(
1133  std::remove_if(variables.begin(), variables.end(), [&variable_names_in_function](const auto& s) { return variable_names_in_function.find(s) == variable_names_in_function.end(); }),
1134  variables.end());
1135  }
1136 
1137  // (4.1) check that the function is not empty, otherwise we return a
1138  // Boolean function with a truth-table with 'X' values
1139  if (this->m_nodes.empty())
1140  {
1141  return OK(std::vector<std::vector<Value>>(1, std::vector<Value>(1 << variables.size(), Value::X)));
1142  }
1143 
1144  // (4.2) safety-check in case the number of variables is too large to process
1145  if (variables.size() > 10)
1146  {
1147  return ERR("could not compute truth table for Boolean function '" + this->to_string() + "': unable to generate truth-table with more than 10 variables");
1148  }
1149 
1150  std::vector<std::vector<Value>> truth_table(this->size(), std::vector<Value>(1 << variables.size(), Value::ZERO));
1151 
1152  // (5) iterate the truth-table rows and set each column accordingly
1153  for (auto value = 0u; value < ((u32)1 << variables.size()); value++)
1154  {
1155  std::unordered_map<std::string, std::vector<Value>> input;
1156  auto tmp = value;
1157  for (const auto& variable : variables)
1158  {
1159  input[variable] = ((tmp & 1) == 0) ? std::vector<Value>({Value::ZERO}) : std::vector<Value>({Value::ONE});
1160  tmp >>= 1;
1161  }
1162  auto result = this->evaluate(input);
1163  if (result.is_error())
1164  {
1165  return ERR(result.get_error());
1166  }
1167  auto output = result.get();
1168  for (auto index = 0u; index < truth_table.size(); index++)
1169  {
1170  truth_table[index][value] = output[index];
1171  }
1172  }
1173 
1174  return OK(truth_table);
1175  }
