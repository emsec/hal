1233  std::vector<std::vector<BooleanFunction::Value>> BooleanFunction::qmc(std::vector<std::vector<Value>> terms)
1234  {
1235  if (terms.empty())
1236  {
1237  return {};
1238  }
1239 
1240  u32 num_variables = terms[0].size();
1241 
1242  // repeatedly merge all groups that only differ in a single value
1243  while (true)
1244  {
1245  bool any_changes = false;
1246  std::vector<std::vector<Value>> new_merged_terms;
1247  std::vector<bool> removed_by_merge(terms.size(), false);
1248  // pairwise compare all terms...
1249  for (u32 i = 0; i < terms.size(); ++i)
1250  {
1251  for (u32 j = i + 1; j < terms.size(); ++j)
1252  {
1253  // ...merge their values and count differences...
1254  std::vector<Value> merged(num_variables, Value::X);
1255  u32 cnt = 0;
1256  for (u32 k = 0; k < num_variables; ++k)
1257  {
1258  if (terms[i][k] == terms[j][k])
1259  {
1260  merged[k] = terms[i][k];
1261  }
1262  else
1263  {
1264  ++cnt;
1265  }
1266  }
1267  // ...and if they differ only in a single value, replace them with the merged term
1268  if (cnt == 1)
1269  {
1270  removed_by_merge[i] = removed_by_merge[j] = true;
1271  new_merged_terms.push_back(merged);
1272  any_changes = true;
1273  }
1274  }
1275  }
1276  if (!any_changes)
1277  {
1278  break;
1279  }
1280  for (u32 i = 0; i < terms.size(); ++i)
1281  {
1282  if (!removed_by_merge[i])
1283  {
1284  new_merged_terms.push_back(terms[i]);
1285  }
1286  }
1287  std::sort(new_merged_terms.begin(), new_merged_terms.end());
1288  new_merged_terms.erase(std::unique(new_merged_terms.begin(), new_merged_terms.end()), new_merged_terms.end());
1289  terms = new_merged_terms;
1290  }
1291 
1292  std::vector<std::vector<Value>> output;
1293 
1294  // build ON-set minterms to later identify essential implicants
1295  std::map<u32, std::vector<u32>> table;
1296  for (u32 i = 0; i < terms.size(); ++i)
1297  {
1298  // find all inputs that are covered by term i
1299  // Example: term=01-1 --> all inputs covered are 0101 and 0111
1300  std::vector<u32> covered_inputs;
1301  for (u32 j = 0; j < num_variables; ++j)
1302  {
1303  if (terms[i][j] != Value::X)
1304  {
1305  if (covered_inputs.empty())
1306  {
1307  covered_inputs.push_back(terms[i][j]);
1308  }
1309  else
1310  {
1311  for (auto& v : covered_inputs)
1312  {
1313  v = (v << 1) | terms[i][j];
1314  }
1315  }
1316  }
1317  else
1318  {
1319  if (covered_inputs.empty())
1320  {
1321  covered_inputs.push_back(0);
1322  covered_inputs.push_back(1);
1323  }
1324  else
1325  {
1326  std::vector<u32> tmp;
1327  for (auto v : covered_inputs)
1328  {
1329  tmp.push_back((v << 1) | 0);
1330  tmp.push_back((v << 1) | 1);
1331  }
1332  covered_inputs = tmp;
1333  }
1334  }
1335  }
1336 
1337  std::sort(covered_inputs.begin(), covered_inputs.end());
1338  covered_inputs.erase(std::unique(covered_inputs.begin(), covered_inputs.end()), covered_inputs.end());
1339 
1340  // now memorize that all these inputs are covered by term i
1341  for (auto v : covered_inputs)
1342  {
1343  table[v].push_back(i);
1344  }
1345  }
1346 
1347  // helper function to add a term to the output and remove it from the table
1348  auto add_to_output = [&](u32 term_index) {
1349  output.push_back(terms[term_index]);
1350  for (auto it2 = table.cbegin(); it2 != table.cend();)
1351  {
1352  if (std::find(it2->second.begin(), it2->second.end(), term_index) != it2->second.end())
1353  {
1354  it2 = table.erase(it2);
1355  }
1356  else
1357  {
1358  ++it2;
1359  }
1360  }
1361  };
1362 
1363  // finally, identify essential implicants and add them to the output
1364  while (!table.empty())
1365  {
1366  bool no_change = true;
1367 
1368  for (auto& it : table)
1369  {
1370  if (it.second.size() == 1)
1371  {
1372  no_change = false;
1373  add_to_output(it.second[0]);
1374  break; // 'add_to_output' invalidates table iterator, so break and restart in next iteration
1375  }
1376  }
1377 
1378  if (no_change)
1379  {
1380  // none of the remaining terms is essential, just pick the one that covers most input values
1381  std::unordered_map<u32, u32> counter;
1382  for (auto it : table)
1383  {
1384  for (auto term_index : it.second)
1385  {
1386  counter[term_index]++;
1387  }
1388  }
1389 
1390  u32 index = std::max_element(counter.begin(), counter.end(), [](const auto& p1, const auto& p2) { return p1.second < p2.second; })->first;
1391  add_to_output(index);
1392  }
1393  }
1394 
1395  // sort output terms for deterministic output order (no impact on functionality)
1396  std::sort(output.begin(), output.end());
1397 
1398  return output;
1399  }
