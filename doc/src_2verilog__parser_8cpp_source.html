656  if (!instance->m_is_module)
657  {
658  instance->m_expanded_port_assignments.clear();

666  m_vcc_gate_types = gate_library->get_vcc_gate_types();

688  }

713  }
714

717  {

719  {

722  Gate* vcc = m_netlist->create_gate(m_netlist->get_unique_gate_id(), vcc_type, "global_vcc");
723

728

730  {
731  return ERR("could not instantiate Verilog netlist '" + m_path.string() + "' with gate library '" + gate_library->get_name() + "': failed to add source to VCC gate");
732  }
733  }
734  else
735  {
736  m_netlist->delete_net(m_one_net);
737  }
738  }
739 
740  // delete unused nets

751  }

762  void VerilogParser::tokenize()
763  {
764  const std::string delimiters = "`,()[]{}\\#*: ;=./";
765  std::string current_token;
766  u32 line_number = 0;
767 
768  std::string line;
769  char prev_char = 0;
770  bool in_string = false;
771  bool escaped = false;
772  bool in_comment = false;
773 
774  std::vector<Token<std::string>> parsed_tokens;
775  while (std::getline(m_fs, line))
776  {
777  line_number++;
778  // this->remove_comments(line, multi_line_comment);
779 
780  for (char c : line)
781  {
782  // deal with comments
783  if (in_comment)
784  {
785  if (c == '/' && prev_char == '*')
786  {
787  in_comment = false;
788  }
789 
790  prev_char = c;
791  continue;
792  }
793 
794  // deal with escaping and strings
795  if (!in_string && c == '\\')
796  {
797  escaped = true;
798  continue;
799  }
800  else if (escaped && std::isspace(c))
801  {
802  escaped = false;
803  continue;
804  }
805  else if (!escaped && c == '"')
806  {
807  in_string = !in_string;
808  }
809 
810  if (!in_comment && ((!std::isspace(c) && delimiters.find(c) == std::string::npos) || escaped || in_string))
811  {
812  current_token += c;
813  }
814  else
815  {
816  // deal with floats
817  if (!current_token.empty())
818  {
819  if (parsed_tokens.size() > 1 && utils::is_digits(parsed_tokens.at(parsed_tokens.size() - 2).string) && parsed_tokens.at(parsed_tokens.size() - 1) == "."
820  && utils::is_digits(current_token))
821  {
822  parsed_tokens.pop_back();
823  parsed_tokens.back() += "." + current_token;
824  }
825  else
826  {
827  parsed_tokens.emplace_back(line_number, current_token);
828  }
829  current_token.clear();
830  }
831 
832  if (!parsed_tokens.empty())
833  {
834  // deal with multi-character tokens
835  if (c == '(' && parsed_tokens.back() == "#")
836  {
837  parsed_tokens.back() = "#(";
838  continue;
839  }
840  else if (c == '*' && parsed_tokens.back() == "(")
841  {
842  parsed_tokens.back() = "(*";
843  continue;
844  }
845  else if (c == ')' && parsed_tokens.back() == "*")
846  {
847  parsed_tokens.back() = "*)";
848  continue;
849  }
850  // start a comment
851  else if (c == '/' && parsed_tokens.back() == "/")
852  {
853  parsed_tokens.pop_back();
854  break;
855  }
856  else if (c == '*' && parsed_tokens.back() == "/")
857  {
858  in_comment = true;
859  parsed_tokens.pop_back();
860  continue;
861  }
862  }
863 
864  if (!std::isspace(c))
865  {
866  parsed_tokens.emplace_back(line_number, std::string(1, c));
867  }
868  }
869  }
870  if (!current_token.empty())
871  {
872  parsed_tokens.emplace_back(line_number, current_token);
873  current_token.clear();
874  }
875  }
876 
877  m_token_stream = TokenStream(parsed_tokens, {"(", "["}, {")", "]"});
878  }

909  Result<std::monostate> VerilogParser::parse_module(std::vector<VerilogDataEntry>& attributes)
910  {
911  std::set<std::string> port_names;
912  std::vector<VerilogDataEntry> internal_attributes;
913 
914  m_token_stream.consume("module", true);
915  const u32 line_number = m_token_stream.peek().number;
916  const std::string module_name = m_token_stream.consume();
917 
918  // verify entity name
919  if (const auto it = m_modules_by_name.find(module_name); it != m_modules_by_name.end())
920  {
921  return ERR("could not parse module '" + module_name + "' (line " + std::to_string(line_number) + "): a module with the same name already exists (line "
922  + std::to_string(it->second->m_line_number) + ")");
923  }
924 
925  auto verilog_module = std::make_unique<VerilogModule>();
926  VerilogModule* verilog_module_raw = verilog_module.get();
927  verilog_module_raw->m_line_number = line_number;
928  verilog_module_raw->m_name = module_name;
929 
930  // parse parameter list
931  if (m_token_stream.consume("#("))
932  {
933  // TODO add support for parameter parsing
934  m_token_stream.consume_until(")");
935  m_token_stream.consume(")", true);
936  log_warning("verilog_parser", "could not parse parameter list provided for module '{}'.", module_name);
937  }
938 
939  // parse port (declaration) list
940  m_token_stream.consume("(", true);
941  Token<std::string> next_token = m_token_stream.peek();
942  if (next_token == "input" || next_token == "output" || next_token == "inout")
943  {
944  if (auto res = parse_port_declaration_list(verilog_module_raw); res.is_error())
945  {
946  return ERR_APPEND(res.get_error(), "could not parse module '" + module_name + "': unable to parse port declaration list (line " + std::to_string(line_number) + ")");
947  }
948  }
949  else
950  {
951  parse_port_list(verilog_module_raw);
952  }
953 
954  m_token_stream.consume(";", true);
955 
956  next_token = m_token_stream.peek();
957  while (next_token != "endmodule")
958  {
959  if (next_token == "input" || next_token == "output" || next_token == "inout")
960  {
961  if (auto res = parse_port_definition(verilog_module_raw, internal_attributes); res.is_error())
962  {
963  return ERR_APPEND(res.get_error(), "could not parse module '" + module_name + "': unable to parse port definition (line " + std::to_string(line_number) + ")");
964  }
965  }
966  else if (next_token == "wire" || next_token == "tri")
967  {
968  if (auto res = parse_signal_definition(verilog_module_raw, internal_attributes); res.is_error())
969  {
970  return ERR_APPEND(res.get_error(), "could not parse module '" + module_name + "': unable to parse signal definition (line " + std::to_string(line_number) + ")");
971  }
972  }
973  else if (next_token == "parameter")
974  {
975  // TODO add support for parameter parsing
976  m_token_stream.consume_until(";");
977  m_token_stream.consume(";", true);
978  log_warning("verilog_parser", "could not parse parameter provided for module '{}'.", module_name);
979  }
980  else if (next_token == "assign")
981  {
982  if (auto res = parse_assignment(verilog_module_raw); res.is_error())
983  {
984  return ERR_APPEND(res.get_error(), "could not parse module '" + module_name + "': unable to parse assignment (line " + std::to_string(line_number) + ")");
985  }
986  }
987  else if (next_token == "(*")
988  {
989  parse_attribute(internal_attributes);
990  }
991  else
992  {
993  if (auto res = parse_instance(verilog_module_raw, internal_attributes); res.is_error())
994  {
995  return ERR_APPEND(res.get_error(), "could not parse module '" + module_name + "': unable to parse instance (line " + std::to_string(line_number) + ")");
996  }
997  }
998 
999  next_token = m_token_stream.peek();
1000  }
1001 
1002  m_token_stream.consume("endmodule", true);
1003 
1004  // assign attributes to entity
1005  if (!attributes.empty())
1006  {
1007  verilog_module->m_attributes.insert(verilog_module->m_attributes.end(), attributes.begin(), attributes.end());
1008  attributes.clear();
1009  }
1010 
1011  // add to collection of entities
1012  m_modules.push_back(std::move(verilog_module));
1013  m_modules_by_name[module_name] = verilog_module_raw;
1014  m_last_module = module_name;
1015 
1016  return OK({});
1017  }

1095  verilog_module->m_ports_by_expression[port->m_expression] = port.get();

1338  Result<std::vector<VerilogDataEntry>> VerilogParser::parse_generic_assign()

1341

1344  {
1345  if (m_token_stream.consume(".", false))
1346  {
1347  line_number = m_token_stream.peek().number;
1348  const Token<std::string> lhs = m_token_stream.join_until("(", "");
1349  m_token_stream.consume("(", true);
1350  const Token<std::string> rhs = m_token_stream.join_until(")", "");
1351  m_token_stream.consume(")", true);
1352 
1353  bool skip = false;
1354  std::string value, data_type;
1355  if (utils::is_integer(rhs.string))
1356  {
1357  value = rhs;
1358  data_type = "integer";
1359  }
1360  else if (utils::is_floating_point(rhs.string))
1361  {
1362  value = rhs;
1363  data_type = "floating_point";
1364  }
1365  else if (rhs.string[0] == '\"' && rhs.string.back() == '\"')
1366  {
1367  value = rhs.string.substr(1, rhs.string.size() - 2);
1368  data_type = "string";
1369  }
1370  else if (isdigit(rhs.string[0]) || rhs.string[0] == '\'')
1371  {

1373  {
1374  skip = true;
1375  }
1376  else
1377  {
1378  value = res.get();
1379  }
1380 
1381  if (value == "0" || value == "1")
1382  {
1383  data_type = "bit_value";
1384  }
1385  else
1386  {
1387  data_type = "bit_vector";
1388  }
1389  }
1390  else
1391  {
1392  return ERR("could not parse generic assignment: failed to identify data type of generic map value '" + rhs.string + "' (line " + std::to_string(line_number) + ")");
1393  }
1394 
1395  if (!skip)
1396  {
1397  generics.push_back(VerilogDataEntry({lhs.string, data_type, value}));
1398  }
1399  }
1400  } while (m_token_stream.consume(",", false));
