302  std::tuple<QList<TreeItem*>, std::pair<bool, int>, bool> ModulePinsTree::getSelectedPins()
303  {
304  QList<TreeItem*> selectedPins; //ordered
305  QSet<TreeItem*> alreadyProcessedPins; //only for performance purposes
306  bool sameGroup = true;
307  bool onlyPins = true;
308  int groupId = -1;
309  for (auto index : selectionModel()->selectedRows())
310  {
311  TreeItem* item = mPortModel->getItemFromIndex(index);
312  auto itemType = mPortModel->getTypeOfItem(item);
313  if (itemType == ModulePinsTreeModel::itemType::pin)
314  {
315  if (!alreadyProcessedPins.contains(item))
316  {
317  selectedPins.append(item);
318  alreadyProcessedPins.insert(item);
319  }
320  }
321  else if (itemType == ModulePinsTreeModel::itemType::group)
322  {
323  onlyPins = false;
324  for (auto pinItem : item->getChildren())
325  {
326  if (!alreadyProcessedPins.contains(pinItem))
327  {
328  selectedPins.append(pinItem);
329  alreadyProcessedPins.insert(pinItem);
330  }
331  }
332  }
333  }
334  //Check if all pins are from the same group (if no pin is selected it is from the same group)
335  if (!selectedPins.isEmpty())
336  {
337  auto mod = gNetlist->get_module_by_id(mModuleID);
338  auto* firstPin = mod->get_pin_by_id(mPortModel->getIdOfItem(selectedPins.front()));
339  groupId = firstPin->get_group().first->get_id();
340  for (auto pinTreeItem : selectedPins)
341  {
342  auto pin = mod->get_pin_by_id(mPortModel->getIdOfItem(pinTreeItem));
343  if (groupId != (int)pin->get_group().first->get_id())
344  {
345  sameGroup = false;
346  break;
347  }
348  }
349  groupId = sameGroup ? groupId : -1;
350  }
351  return std::make_tuple(selectedPins, std::make_pair(sameGroup, groupId), onlyPins);
352  }
