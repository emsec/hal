314  Result<std::monostate> move_pin(T* pin, u32 new_index)
315  {
316  if (pin == nullptr)
317  {
318  return ERR("'nullptr' given instead of a pin when trying to move pin within pin group '" + m_name + "' with ID " + std::to_string(m_id));
319  }
320 
321  if (pin->m_group.first != this)
322  {
323  return ERR("pin '" + pin->get_name() + "' with ID " + std::to_string(pin->get_id()) + " does not belong to pin group '" + m_name + "' with ID " + std::to_string(m_id));
324  }
325 
326  if (new_index >= m_start_index && new_index < m_next_index)
327  {
328  u32 old_index = pin->m_group.second;
329  if (old_index == new_index)
330  {
331  return OK({});
332  }
333 
334  i32 direction = (old_index > new_index) ? 1 : -1;
335  m_pins.erase(std::next(m_pins.begin(), old_index - m_start_index));
336  auto it = std::next(m_pins.begin(), new_index - m_start_index);
337  it = m_pins.insert(it, pin);
338  for (u32 i = new_index; i != old_index; i += direction)
339  {
340  std::advance(it, direction);
341  std::get<1>((*it)->m_group) += direction;
342  }
343  std::get<1>(pin->m_group) = new_index;
344 
345  return OK({});
346  }
347  else
348  {
349  return ERR("new index (" + std::to_string(new_index) + ") for pin '" + pin->get_name() + "' with ID " + std::to_string(pin->get_id()) + " is not between start index ("
350  + std::to_string(m_start_index) + ") and end index(" + std::to_string(m_next_index - 1) + ") of pin group '" + m_name + "' with ID " + std::to_string(m_id));
351  }
352  }

365  }

370  }
371

373  pin->m_group = std::make_pair(nullptr, 0);
374  auto it = std::next(m_pins.begin(), index - m_start_index);
375  it = m_pins.erase(it);

377  {
