62  static void handleProgramArguments(const ProgramArguments& args)
63  {
64  enum OpenArgs { None = 0, OpenProject = 1, DefaultImport = 2, ImportToProject = 3 } openArgs = None;
65  ProjectManager* pm = ProjectManager::instance();
66  std::filesystem::path fileName;
67  ProjectDirectory projDir;
68  QString gateLibraryPath;
69  if (args.is_option_set("--project-dir"))
70  {
71  openArgs = (OpenArgs) (openArgs|OpenProject);
72  projDir = ProjectDirectory(args.get_parameter("--project-dir"));
73  log_info("gui", "GUI started with project {}.", projDir.string());
74  }
75  if (args.is_option_set("--import-netlist"))
76  {
77  openArgs = (OpenArgs) (openArgs|DefaultImport);
78  fileName = std::filesystem::path(args.get_parameter("--import-netlist"));
79  log_info("gui", "GUI started with file {}.", fileName.string());
80  if (args.is_option_set("--gate-library"))
81  gateLibraryPath = QString::fromStdString(args.get_parameter("--gate-library"));
82  }
83  switch (openArgs)
84  {
85  case None:
86  return;
87  case OpenProject:
88  FileManager::get_instance()->openProject(QString::fromStdString(projDir.string()));
89  break;
90  case DefaultImport:
91  projDir = ProjectDirectory(fileName);
92  [[fallthrough]];
93  // continue with Import
94  case ImportToProject:
95  if (!pm->create_project_directory(projDir.string()))
96  {
97  log_error("gui", "Cannot create project directory {}.", projDir.string());
98  return;
99  }
100  else
101  {
102  std::filesystem::path lpath = pm->get_project_directory().get_default_filename(".log");
103  LogManager::get_instance()->set_file_name(lpath);
104  if (!FileManager::get_instance()->deprecatedOpenFile(QString::fromStdString(fileName.string()),gateLibraryPath))
105  log_error("gui", "Failed to open netlist '{}'.", fileName.string());
106  }
107  break;
108  }
109  }
