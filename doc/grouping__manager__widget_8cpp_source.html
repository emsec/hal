315  void GroupingManagerWidget::newGroupingSuccOrPred(int maxDepth, bool succ, const GraphicsItem* item)
316  {
317  QSet<u32> mods, gats, nets;
318  ToolboxNode tbn;
319  if (item)
320  tbn = ToolboxNode(item);
321  ToolboxModuleHash tmh(tbn.mNode);
322  QVector<Net*> todoNet;
323  QSet<Node> handledBox;
324  QSet<Net*> handledNet;
325  todoNet.append(QVector<Net*>::fromStdVector(succ ? tbn.outputNets() : tbn.inputNets()));
326  handledBox.insert(tbn.mNode);
327  for (int loop = 0; !maxDepth || loop < maxDepth; loop++)
328  {
329  if (todoNet.isEmpty())
330  break;
331  QVector<Net*> nextRound;
332  for (Net* n : todoNet)
333  {
334  if (handledNet.contains(n))
335  continue;
336  handledNet.insert(n);
337  nets.insert(n->get_id());
338  for (Endpoint* ep : (succ ? n->get_destinations() : n->get_sources()))
339  {
340  ToolboxNode nextNode(ep, &tmh);
341  if (handledBox.contains(nextNode.mNode))
342  continue;
343  handledBox.insert(nextNode.mNode);
344  switch (nextNode.mNode.type())
345  {
346  case Node::Module:
347  mods.insert(nextNode.mNode.id());
348  break;
349  case Node::Gate:
350  gats.insert(nextNode.mNode.id());
351  break;
352  default:
353  continue;
354  }
355  nextRound.append(QVector<Net*>::fromStdVector(succ ? nextNode.outputNets() : nextNode.inputNets()));
356  }
357  }
358  todoNet = nextRound;
359  }
360 
361  UserActionCompound* act = new UserActionCompound;
362  act->setUseCreatedObject();
363  act->addAction(new ActionCreateObject(UserActionObjectType::Grouping, tbn.mName + (succ ? " successor" : " predecessor")));
364  act->addAction(new ActionAddItemsToObject(mods, gats, nets));
365  act->exec();
366  }

378  void GroupingManagerWidget::newGroupingByDistance(int maxDepth, bool succ, const GraphicsItem* item)
379  {
380  ToolboxNode tbn;
381  if (item)
382  tbn = ToolboxNode(item);
383  ToolboxModuleHash tmh(tbn.mNode);
384  QVector<Net*> thisDst;
385  QVector<Net*> nextDst;
386  QSet<Node> handledNode;
387  QSet<Net*> handledNet;
388  nextDst.append(QVector<Net*>::fromStdVector(succ ? tbn.outputNets() : tbn.inputNets()));
389  handledNode.insert(tbn.mNode);
390  for (int iround = 1; iround <= maxDepth; iround++)
391  {
392  if (nextDst.isEmpty())
393  break;
394  thisDst = nextDst;
395  nextDst.clear();
396  QSet<u32> mods, gats, nets;
397  for (Net* n : thisDst)
398  {
399  if (handledNet.contains(n))
400  continue;
401  handledNet.insert(n);
402  nets.insert(n->get_id());
403  for (Endpoint* ep : succ ? n->get_destinations() : n->get_sources())
404  {
405  ToolboxNode nextNode(ep, &tmh);
406  if (handledNode.contains(nextNode.mNode))
407  continue;
408  handledNode.insert(nextNode.mNode);
409  switch (nextNode.mNode.type())
410  {
411  case Node::Module:
412  mods.insert(nextNode.mNode.id());

418  continue;
419  }
