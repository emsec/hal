311  m_plugin_options.remove(flag);
312  }
313  }
314  }
315  {
316  auto tmp = m_cli_option_to_ui_plugin_name;
317  for (auto [flag, name] : tmp)
318  {
319  if (name == plugin_name)

321  m_cli_option_to_ui_plugin_name.erase(flag);
322  m_plugin_options.remove(flag);
323  }
324  }

328  auto plugin_inst = std::move(std::get<0>(it->second));
329  m_loaded_plugins.erase(it);

332  plugin_inst->on_unload();
