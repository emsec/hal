141  void CodeEditor::highlightCurrentLine()
142  {
143  /*
144  * Qt doesn't want to highlight blocks in line-wrap mode (even if the docs
145  * say so), so we hack our way around it by going to the end of a block and
146  * then go upwards selecting each line in that block separately.
147  */
148  QList<QTextEdit::ExtraSelection> extra_selections;
149  QTextCursor cursor(textCursor());
150  cursor.clearSelection();
151  cursor.movePosition(QTextCursor::EndOfBlock);
152  cursor.movePosition(QTextCursor::StartOfLine);
153  int block = cursor.blockNumber();
154  int oldPosition;
155 
156  do
157  {
158  QTextEdit::ExtraSelection selection;
159 
160  selection.format.setBackground(mCurrentLineBackground);
161  selection.format.setProperty(QTextFormat::FullWidthSelection, true);
162  selection.cursor = cursor;
163  extra_selections.append(selection);
164 
165  oldPosition = cursor.position();
166  cursor.movePosition(QTextCursor::Up);
167  // stop when we leave the block or hit the top of the document
168  } while(cursor.blockNumber() == block && cursor.position() != oldPosition);
169 
170  setExtraSelections(extra_selections);
171  }
